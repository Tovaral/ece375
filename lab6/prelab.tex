            out TCNT1L, R24

        \end{verbatim}

   \item
        \begin{verbatim}
            in R24, LOW(TCNT1)
            in R25, HIGH(TCNT1)
        \end{verbatim}
    \item $0xFF-127=128$ 'ticks' of the timer will occur before the counter overflows. 1 Hertz is 1 cycle per second so 1 Mhz is $10^6$ cycles per second. $128/(16 \times 10^6)=8e-06$ seconds which is 8 microseconds.
\end{enumerate}
